MSG_TYPE_VEHICLE = 0x01

STRUCT_MC = '<BfBffffffBBffBf'

# >>> AGGIUNGI QUESTE RIGHE <<<
# Mode definitions
MODE_ANGLES = 0
MODE_LONG = 1
MODE_RANGE = 2  # Solo range_3d disponibile, ENU/angles = 0

# SOSTITUISCI la funzione decode_mc_message con questa versione:

def decode_mc_message(payload_b64):
    """Decodifica messaggio mission control."""
    try:
        data = base64.b64decode(payload_b64)
        values = struct.unpack(STRUCT_MC, data)
        
        if values[0] != MSG_TYPE_MC:
            return None
        
        mode = values[2]
        
        # >>> MODIFICA QUI: Gestisci MODE_RANGE <<<
        if mode == 1:
            mode_str = 'LONG'
        elif mode == 2:
            mode_str = 'RANGE'  # NUOVO
        else:
            mode_str = 'ANGLES'
        
        return {
            'timestamp': values[1],
            'usbl_data': {
                'mode': mode_str,  # Ora può essere 'ANGLES', 'LONG', o 'RANGE'
                'enu': {'east_m': values[3], 'north_m': values[4], 'up_m': values[5]},
                'angles': {
                    'bearing_rad': values[6],
                    'elevation_rad': values[7],
                    'bearing_deg': math.degrees(values[6]),
                    'elevation_deg': math.degrees(values[7])
                },
                'estimated_distance_m': values[8],
                'integrity': values[9],
                'range_3d_m': values[14]
            },
            'control_data': {
                'enable': values[10] == 1,
                'bias_angle_deg': values[11],
                'frequency_hz': values[12],
                'use_range_only': values[13] == 1
            }
        }
    except Exception:
        return None

# SOSTITUISCI la funzione get_enu() con questa versione:

def get_enu(self):
    """
    Ottiene posizione ENU dal mission control.
    Gestisce i diversi mode: LONG, ANGLES, RANGE.
    """
    if not self.last_mc_data['valid']:
        return (0.0, 0.0, 0.0)
    
    usbl = self.last_mc_data['usbl_data']
    mode = usbl.get('mode', 'ANGLES')
    
    # Mode LONG: usa direttamente ENU
    if mode == 'LONG':
        return (usbl['enu']['east_m'], usbl['enu']['north_m'], usbl['enu']['up_m'])
    
    # Mode ANGLES: calcola ENU da bearing/elevation
    elif mode == 'ANGLES':
        bearing = usbl['angles']['bearing_rad']
        elevation = usbl['angles']['elevation_rad']
        distance = usbl['range_3d_m']
        
        e = distance * math.cos(elevation) * math.sin(bearing)
        n = distance * math.cos(elevation) * math.cos(bearing)
        u = distance * math.sin(elevation)
        
        return (e, n, u)
    
    # Mode RANGE: solo range_3d disponibile, ENU non calcolabile
    else:  # mode == 'RANGE'
        # Ritorna (0, 0, 0) perché non abbiamo angoli per calcolare ENU
        return (0.0, 0.0, 0.0)
